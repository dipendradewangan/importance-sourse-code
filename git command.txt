git init //for initialize git folder 

git add fileName/.  // for adding git 

git commit -m "message"  // for commit

git remote add origin "url of reposatory" // for select remote reposatory

git push origin master // for push selected reposatory in master branch

git branch newBranch // for make new branch

git checkout -b newBranch // for create and switch to new branch

git branch -a // for checking branch list

git checkout branchName // for switch to another branch

git push origin --delete branchName // for deleting another branch in remote

git branch -d branchName // for deleting another branch in local pc
